        function loadData(){
            var xhr = new XMLHttpRequest();
            xhr.open("GET", "contact.dat", true);

            xhr.onreadystatechange = function(){
                if(xhr.readyState == 4 && xhr.status == 200){
                    var data = xhr.responseText.trim().split("\n");
                    var table = "<table border='1'><tr><th>SrNo</th><th>Name</th><th>Res</th><th>Mobile</th><th>Address</th></tr>";

                    for(var i=0; i<data.length; i++){
                        var row = data[i].split(",");
                        table += "<tr>";
                        for(var j=0; j<row.length; j++){
                            table += "<td>" + row[j] + "</td>";
                        }
                        table += "</tr>";
                    }

                    table += "</table>";
                    document.getElementById("result").innerHTML = table;
                }
            };
            xhr.send();
        }
    </script>

import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression

# Create dataset
data = {
    'Height': [150,155,160,165,170,175,180,185],
    'Weight': [50,55,60,65,70,75,80,85]
}

df = pd.DataFrame(data)

# Independent and target variables
X = df[['Height']]
y = df['Weight']

# Split dataset (70:30)
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=0)

# Print splits
print("X_train:\n", X_train)
print("\nX_test:\n", X_test)
print("\ny_train:\n", y_train)
print("\ny_test:\n", y_test)

# Build model
model = LinearRegression()
model.fit(X_train, y_train)

# Predictions
y_pred = model.predict(X_test)

print("\nPredicted Weights:\n", y_pred)
