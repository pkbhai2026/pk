import pandas as pd
from mlxtend.preprocessing import TransactionEncoder
from mlxtend.frequent_patterns import apriori, association_rules

# Custom dataset
dataset = [
    ['Rice','Dal','Oil'],
    ['Rice','Sugar'],
    ['Dal','Oil'],
    ['Rice','Dal'],
    ['Sugar','Oil'],
    ['Rice','Oil','Sugar']
]

# Convert categorical to numeric
te = TransactionEncoder()
te_data = te.fit(dataset).transform(dataset)
df = pd.DataFrame(te_data, columns=te.columns_)

print("Encoded Dataset:\n", df)

# Apply Apriori
freq1 = apriori(df, min_support=0.5, use_colnames=True)
rules1 = association_rules(freq1, metric="confidence", min_threshold=0.6)

print("\nFrequent Itemsets:\n", freq1)
print("\nAssociation Rules:\n", rules1)

# Different min_support
freq2 = apriori(df, min_support=0.3, use_colnames=True)
rules2 = association_rules(freq2, metric="confidence", min_threshold=0.6)

print("\nFrequent Itemsets (min_sup=0.3):\n", freq2)
print("\nAssociation Rules:\n", rules2)
