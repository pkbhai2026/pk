        function showBook(str){
            if(str == ""){
                document.getElementById("result").innerHTML = "";
                return;
            }

            var xhr = new XMLHttpRequest();
            xhr.open("GET", "books.xml", true);

            xhr.onreadystatechange = function(){
                if(xhr.readyState == 4 && xhr.status == 200){
                    var xml = xhr.responseXML;
                    var books = xml.getElementsByTagName("book");
                    var output = "";

                    for(var i=0; i<books.length; i++){
                        var title = books[i].getElementsByTagName("title")[0].childNodes[0].nodeValue;

                        if(title == str){
                            var author = books[i].getElementsByTagName("author")[0].childNodes[0].nodeValue;
                            var year = books[i].getElementsByTagName("year")[0].childNodes[0].nodeValue;
                            var price = books[i].getElementsByTagName("price")[0].childNodes[0].nodeValue;

                            output = "Author: " + author + "<br>Year: " + year + "<br>Price: " + price;
                        }
                    }

                    document.getElementById("result").innerHTML = output;
                }
            };
            xhr.send();
        }

import re
from sklearn.feature_extraction.text import CountVectorizer
import numpy as np

# Sample paragraph
text = """
Machine learning is a method of data analysis that automates analytical model building.
It is a branch of artificial intelligence based on the idea that systems can learn from data,
identify patterns and make decisions with minimal human intervention.
"""

# Preprocessing: remove special characters and digits
clean_text = re.sub(r'[^a-zA-Z\s]', '', text)

# Sentence tokenization
sentences = clean_text.split('.')

# Vectorization
vectorizer = CountVectorizer().fit_transform(sentences)
vectors = vectorizer.toarray()

# Sentence scoring (simple sum of word frequencies)
scores = vectors.sum(axis=1)

# Get top 2 sentences
top_sentences = np.argsort(scores)[-2:]

# Generate summary
summary = " ".join([sentences[i] for i in top_sentences])

print("Clean Text:\n", clean_text)
print("\nSummary:\n", summary)
