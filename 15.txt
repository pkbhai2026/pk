        function showSuggestions(str){
            if(str.length == 0){
                document.getElementById("result").innerHTML = "";
                return;
            }

            var xhr = new XMLHttpRequest();
            xhr.open("GET", "suggest.php?q=" + str, true);

            xhr.onreadystatechange = function(){
                if(xhr.readyState == 4 && xhr.status == 200){
                    document.getElementById("result").innerHTML = xhr.responseText;
                }
            };
            xhr.send();
        }
    </script>
</head>

Enter Name: <input type="text" onkeyup="showSuggestions(this.value)">

$suggestions = ["Apple","Banana","Mango","Orange","Grapes","Pineapple","Papaya"];

$q = $_GET['q'];
$hint = "";

if($q != ""){
    $q = strtolower($q);
    foreach($suggestions as $name){
        if(str_starts_with(strtolower($name), $q)){
            $hint .= $name . "<br>";
        }
    }
}

echo $hint == "" ? "No suggestions" : $hint;

import pandas as pd
from mlxtend.preprocessing import TransactionEncoder
from mlxtend.frequent_patterns import apriori, association_rules

# Sample dataset
dataset = [
    ['Milk','Bread','Butter'],
    ['Bread','Butter'],
    ['Milk','Bread'],
    ['Milk','Butter'],
    ['Bread','Eggs'],
    ['Milk','Eggs'],
    ['Bread','Butter','Eggs']
]

# Convert categorical to numeric
te = TransactionEncoder()
te_data = te.fit(dataset).transform(dataset)
df = pd.DataFrame(te_data, columns=te.columns_)

print("Encoded Dataset:\n", df)

# Apriori with min_support = 0.5
freq1 = apriori(df, min_support=0.5, use_colnames=True)
rules1 = association_rules(freq1, metric="confidence", min_threshold=0.7)

print("\nFrequent Itemsets (min_sup=0.5):\n", freq1)
print("\nAssociation Rules:\n", rules1)

# Apriori with min_support = 0.3
freq2 = apriori(df, min_support=0.3, use_colnames=True)
rules2 = association_rules(freq2, metric="confidence", min_threshold=0.7)

print("\nFrequent Itemsets (min_sup=0.3):\n", freq2)
print("\nAssociation Rules:\n", rules2)
